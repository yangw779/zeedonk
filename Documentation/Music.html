function update(){
  var mx = Mouse.x;
  var my = Mouse.y;
  var cx = Gfx.screenwidthmid;
  var cy = Gfx.screenheightmid;
  var dx = mx-cx;
  var dy = my-cy;
  var d = Math.sqrt(dx*dx+dy*dy);
  //gets smaller the further away you are
  var pc = Math.pow(1/(1+d),1/2);
  Music.musicvol = pc;
  Gfx.drawline(cx,cy,mx,my,Col.RED);
}
